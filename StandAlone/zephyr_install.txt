# From codespaces
# 1. Deps (ninja & dtc only — cmake and pip already present in Codespace)
sudo apt-get install -y --no-install-recommends ninja-build device-tree-compiler
pip3 install kconfiglib --break-system-packages

# 2. SDK — minimal scaffold + riscv toolchain only
wget https://github.com/zephyrproject-rtos/sdk-ng/releases/download/v1.0.1/zephyr-sdk-1.0.1_linux-x86_64_minimal.tar.xz
wget https://github.com/zephyrproject-rtos/sdk-ng/releases/download/v1.0.1/toolchain_gnu_linux-x86_64_riscv64-zephyr-elf.tar.xz

tar xf zephyr-sdk-1.0.1_linux-x86_64_minimal.tar.xz -C /opt
tar xf toolchain_gnu_linux-x86_64_riscv64-zephyr-elf.tar.xz -C /opt/zephyr-sdk-1.0.1

# 3. Fix SDK 1.0.1 layout bug — cmake expects gnu/ subdir
mkdir -p /opt/zephyr-sdk-1.0.1/gnu
ln -s /opt/zephyr-sdk-1.0.1/riscv64-zephyr-elf /opt/zephyr-sdk-1.0.1/gnu/riscv64-zephyr-elf

# 4. Environment
export ZEPHYR_SDK_INSTALL_DIR=/opt/zephyr-sdk-1.0.1
export ZEPHYR_BASE=~/zephyrproject/zephyr
export ZEPHYR_TOOLCHAIN_VARIANT=zephyr

# 5. Build
cmake -B build -GNinja \
  -DBOARD=qemu_riscv32 \
  -DCONFIG_BUILD_OUTPUT_BIN=y \
  -DCONFIG_MINIMAL_LIBC=y \
  ~/zephyrproject/zephyr/samples/hello_world

ninja -C build

ls build/zephyr/zephyr.bin
